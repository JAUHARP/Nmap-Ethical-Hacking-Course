$sudo nmap -sA scanme.nmap.org

# cmd for connect scan
$nmap -sT scanme.nmap.org

#cmd to scan top 1000 ports by default
$sudo nmap -sT scanme.nmap.org

#cmd for tcp null scan
$sudo nmap -sN scanme.nmap.org

#cmd to scan connect top 100 ports
$sudo nmap -sT scanme.nmap.org -F

#cmd udp scan with specified ports
$sudo nmap -sU -p19,53,123 scanme.nmap.org

#cmd for tcp xmas scan
sudo nmap -sX scanme.nmap.org
